## AQL
```aql
// make items the source table of sum() because it is on the many side (an order has many items, but an item has only 1 order = 1 discount)
metric order_revenue = items | sum(items.price - orders.discount);
metric avg_order_value_per_city =
  unique(city.id)
  | select(value: order_revenue)
  | avg(value);

explore {
  measures {
    avg_order_value_per_city: avg_order_value_per_city,
  }
  sorts {
    avg_order_value_per_city DESC,
  }
}
```
