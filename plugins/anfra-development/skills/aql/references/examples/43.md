## AQL
```aql
metric user_count = users | count(users.id);
metric cities_with_more_than_100_users = unique(public_cities.id) | filter(user_count > 100);

explore {
  dimensions {
    country: countries.name,
  }
  measures {
    count_cities_with_more_than_100_users: cities_with_more_than_100_users | count(),
  }
  filters {
    public_cities.id in cities_with_more_than_100_users,
  }
  sorts {
    country ASC,
  }
}
```
