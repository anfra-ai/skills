## AQL
```aql
metric order_count = orders | count(orders.id);
metric total_order_count = order_count | exclude_grains(products.id, products.name, products.created_at);
metric order_percentage = order_count * 100.0 / total_order_count;
explore {
  dimensions {
    product_id: products.id,
    product_name: products.name,
    created_at: products.created_at,
  }
  measures {
    order_count: order_count,
    total_order_count: total_order_count,
    percentage: order_percentage,
  }
  sorts {
    created_at DESC,
    product_name ASC,
  }
}
```
