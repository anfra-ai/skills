## AQL
```aql
metric order_rank_by_time_per_user = rank(
  order: orders.created_at | day() | asc(),
  partition: users.id,
);
metric first_order_date =
  orders // MUST use the many-side as source table
  | filter(order_rank_by_time_per_user == 1) // MUST use filter() instead of where(), because we're filtering on metric
  | max(orders.created_at | day());
metric third_order_date =
  orders
  | filter(order_rank_by_time_per_user == 3)
  | max(orders.created_at | day());

explore {
  dimensions {
    user_id: users.id,
    user_name: users.name,
  }
  measures {
    first_order_date: first_order_date,
    third_order_date: third_order_date,
  }
  sorts {
    user_name ASC,
  }
}
```
