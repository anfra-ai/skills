## AQL
```aql
metric user_count = users | count(users.id);
metric user_count_last_month = user_count | where(users.created_at is @(last month));
metric country_rank_by_user_count_last_month = rank(order: user_count_last_month | desc());
explore {
  dimensions {
    country_name: countries.name,
  }
  measures {
    user_count: user_count_last_month,
    country_rank_by_user_count_last_month: country_rank_by_user_count_last_month,
  }
  filters {
    country_rank_by_user_count_last_month < 4,
  }
  sorts {
    user_count DESC,
    country_name ASC,
  }
}
```
