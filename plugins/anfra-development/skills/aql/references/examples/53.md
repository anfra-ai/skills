## AQL
```aql
metric order_rank_by_time_per_user = rank(
  order: orders.created_at | asc(),
  partition: users.id,
);
metric total_price_first_3_orders =
  items // MUST use the many-side as source table
  | filter(order_rank_by_time_per_user < 4) // MUST use filter() instead of where(), because we're filtering on metric
  | sum(items.price);
metric total_price_all_orders = items | sum(items.price);

explore {
  dimensions {
    user_id: users.id,
    user_name: users.name,
  }
  measures {
    total_price_first_3_orders: total_price_first_3_orders,
    total_price_all_orders: total_price_all_orders,
  }
  sorts {
    user_name ASC,
  }
}
```
