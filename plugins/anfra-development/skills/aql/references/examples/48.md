## AQL
```aql
dimension users.age = date_diff('year', users.birth_date, @(today));
metric rank_by_avg_user_age_in_each_country = rank(order: users | avg(users.age) | desc(), partition: countries.name);
metric top_3_oldest_cities_per_country =
  unique(public_cities.id, countries.name)
  | filter(rank_by_avg_user_age_in_each_country < 4);
metric order_revenue = items | sum(items.price - orders.discount);
metric avg_order_revenue_per_city =
  unique(public_cities.id) // NOTE: calculate order_revenue of each city
  | select(v: order_revenue)
  | avg(v);

explore {
  // NOTE: do not include cities or users in the dimensions. We want to break down by country only
  dimensions {
    country: countries.name,
  }
  measures {
    avg_order_revenue_per_city: avg_order_revenue_per_city,
  }
  filters {
    public_cities.id in (top_3_oldest_cities_per_country | select(public_cities.id)),
  }
  sorts {
    country ASC,
  }
}
```
