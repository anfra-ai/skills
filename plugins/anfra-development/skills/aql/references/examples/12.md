dimension public_cities.order_count = users.order_count | dimensionalize(public_cities.id);
dimension countries.order_count = users.order_count | dimensionalize(countries.id);
explore {
  dimensions {
    user_id: users.id,
    user_name: users.name,
    user_created_at: users.created_at,
    city_order_count: public_cities.order_count,
    country_order_count: countries.order_count,
  }
  measures {
    user_order_count: users.order_count,
  }
  sorts {
    user_created_at DESC,
  }
}
