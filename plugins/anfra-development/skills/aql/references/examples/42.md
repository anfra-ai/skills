## AQL
```aql
metric user_count = users | count(users.id);
metric user_count_last_month = user_count | where(users.created_at is @(last month));
metric countries_with_more_than_100_users_created_last_month = unique(countries.name) | filter(user_count_last_month > 100);

explore {
  dimensions {
    city_id: public_cities.id,
    city_name: public_cities.name,
  }
  measures {
    user_count_last_month: user_count_last_month,
  }
  filters {
    countries.name in countries_with_more_than_100_users_created_last_month,
  }
  sorts {
    city_id ASC,
  }
}
```
