## AQL
```aql
metric user_count = users | count(users.id);
metric monthly_avg_user_sign_ups =
  unique(users.sign_up_at | month())
  | select(user_count: user_count)
  | avg(user_count);

explore {
  dimensions {
    sign_up_year: users.sign_up_at | year(),
  }
  measures {
    monthly_avg_user_sign_ups: monthly_avg_user_sign_ups,
  }
  sorts {
    sign_up_year ASC,
  }
}
```
