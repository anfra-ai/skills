## AQL
```aql
dimension users.age = date_diff('year', users.birth_date, @(today));
metric rank_by_user_age_in_each_city = rank(order: users | max(users.age) | desc(), partition: public_cities.name);
metric top_3_oldest_users_per_city =
  unique(users.id, public_cities.name)
  | filter(rank_by_user_age_in_each_city < 4);
metric order_revenue = items | sum(items.price - orders.discount);
metric avg_order_revenue_per_user =
  unique(users.id) // NOTE: calculate order_revenue of each user
  | select(v: order_revenue)
  | avg(v);

explore {
  // NOTE: do not include users in the dimensions. Otherwise, the measures would be broken down by users, while we want to break down by city only
  dimensions {
    city_name: public_cities.name,
  }
  measures {
    total_order_revenue: order_revenue,
    avg_order_revenue_per_user: avg_order_revenue_per_user,
  }
  filters {
    users.id in (top_3_oldest_users_per_city | select(users.id)),
  }
  sorts {
    city_name ASC,
  }
}
```
