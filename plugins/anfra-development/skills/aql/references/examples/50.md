## AQL
```aql
dimension users.age = date_diff('year', users.birth_date, @(today));
metric order_revenue = items | sum(items.price - orders.discount);
metric order_count = orders | count(1);
metric rank_by_revenue_in_each_country = rank(order: order_revenue | desc(), partition: countries.name);
metric top_3_cities_per_country =
  unique(public_cities.id, countries.name)
  | filter(rank_by_revenue_in_each_country < 4);
metric avg_order_count_of_cities =
  unique(public_cities.id) // NOTE: calculate order_count of each city
  | where(public_cities.id in (top_3_cities_per_country | select(public_cities.id)))
  | select(v: order_count)
  | avg(v);

explore {
  // NOTE: do not include cities in the dimensions. We want to break down by country only
  dimensions {
    country: countries.name,
  }
  measures {
    avg_order_count_of_cities: avg_order_count_of_cities,
  }
  filters {
  }
  sorts {
    country ASC,
  }
}
```
