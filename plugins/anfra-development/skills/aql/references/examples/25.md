## Dataset fields
```yaml
dimensions:
- field: id
  model: orders
- field: discount
  model: orders
- field: created_at
  model: orders
measures:
- field: value
  model: orders
```

## AQL
```aql
metric total_discount = orders | sum(orders.discount);
metric revenue =
  unique(orders.id)
  | select(
    revenue_per_order: orders.value - total_discount,
  )
  | sum(revenue_per_order);

explore {
  dimensions {
    time: orders.created_at | month(),
  }
  measures {
    revenue: revenue,
  }
  sorts {
    time ASC,
  }
}
```
