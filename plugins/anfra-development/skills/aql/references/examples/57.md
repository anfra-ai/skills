## AQL
```aql
metric previous_review_at = previous(reviews.created_at, order: reviews.id | asc(), partition: reviews.user_id);
metric avg_days_between_reviews =
	reviews
  | select(
		review_at: reviews.created_at,
		previous_review_at: previous_review_at,
  )
	| select(
		days_between: date_diff('day', previous_review_at, review_at),
  )
	| avg();
explore {
  dimensions {
    user_id: users.id,
    user_name: users.name,
		avg_days_between_reviews: avg_days_between_reviews,
  }
  sorts {
    avg_days_between_reviews DESC,
		user_id ASC,
  }
}
```
