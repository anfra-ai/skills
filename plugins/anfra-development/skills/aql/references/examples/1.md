## AQL
```aql
metric user_count = users | count(users.id);
metric user_count_by_city =
  unique(cities.id)
  | select(user_count: user_count);
metric avg_city_user_count = user_count_by_city | avg(user_count);

explore {
  dimensions {
    country: countries.name,
  }
  measures {
    avg_city_user_count: avg_city_user_count,
  }
  sorts {
    country ASC,
  }
}
```
