## AQL
```aql
metric order_count = orders | count(orders.id);
metric highest_order_count_by_user =
  unique(db_users.id)
  | select(order_count: order_count)
  | max(order_count);

explore {
  dimensions {
    year: orders.created_at | year(),
  }
  measures {
    highest_order_count_by_user: highest_order_count_by_user,
  }
  filters {
    orders.created_at is not @(2020),
  }
  sorts {
    year ASC,
  }
}
```
