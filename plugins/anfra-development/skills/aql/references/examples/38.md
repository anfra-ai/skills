## AQL
```aql
metric product_rank =
  product
  | where(product.price > 100)
  | rank(order: product | max(product.price) | desc());

explore {
  dimensions {
    product_id: product.id,
    product_name: product.name,
    product_price: product.price,
  }
  measures {
    order_count: order_count,
    product_rank: product_rank,
  }
  filters {
    product_rank < 5,
  }
  sorts {
    order_count DESC,
    product_id ASC,
  }
}
```
